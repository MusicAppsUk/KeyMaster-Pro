name: Generate Jack voice pack
on:
  workflow_dispatch:
    inputs:
      dry_run:
        description: "true = preview (no cost). false = generate the 17-line Jack pack."
        default: "true"
permissions:
  contents: write
jobs:
  generate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: "20"
      - name: Generate or preview
        env:
          ELEVENLABS_API_KEY: ${{ secrets.ELEVENLABS_API_KEY }}
          DRY: ${{ inputs.dry_run }}
        run: |
          A=(--voice-id RL2gbGArFsmr05q4aJLj --out voice/en-GB)
          if [ "$DRY" = "false" ]; then A+=(--generate); fi
          node tools/generateVoicePack.mjs "${A[@]}"
      - name: Publish Jack to test branch
        if: ${{ inputs.dry_run == 'false' }}
        run: |
          git config user.name bot
          git config user.email bot@users.noreply.github.com
          git checkout -B voice-pack-jack
          sed -i -E 's/(PREMIUM_VOICE_READY[[:space:]]*=[[:space:]]*)false/\1true/' foundations.js || true
          git add voice/en-GB/*.mp3 foundations.js tools/voice-pack.lock.json
          git commit -m "Jack voice pack" || echo nothing
          git push -f origin voice-pack-jack

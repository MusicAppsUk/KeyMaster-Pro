name: Generate voice pack

# Manual only. Generates KeyMaster PRO tutor narration via ElevenLabs using a
# PRIVATE repository secret (ELEVENLABS_API_KEY). The key is never printed and
# never committed. The app itself never calls ElevenLabs — it only plays the
# resulting local MP3s. Default voice: Jack (RL2gbGArFsmr05q4aJLj).
#
# SAFETY: the first run is a DRY RUN by default (no API call, no spend). To
# actually generate, re-run with dry_run = false. The job refuses to render more
# than 2 lines unless you explicitly approve.

on:
  workflow_dispatch:
    inputs:
      voice_id:
        description: "ElevenLabs voice ID"
        required: false
        default: "RL2gbGArFsmr05q4aJLj"   # Jack
      only:
        description: "Line IDs to generate (comma-separated). Default = the 2-line proof."
        required: false
        default: "welcome.0,find-c.0"
      limit:
        description: "Optional extra cap on number of lines (blank = none)."
        required: false
        default: ""
      dry_run:
        description: "Dry run? true = plan only, no API call, no files. false = generate."
        required: false
        default: "true"
      publish:
        description: "How to deliver files: 'artifact' (safe, downloadable) or 'branch' (commit to voice-pack-proof)."
        required: false
        default: "artifact"
      out_dir:
        description: "Output dir the app serves (beside index.html)."
        required: false
        default: "voice/en-GB"
      approve_more:
        description: "Type 'I APPROVE MORE' to allow more than 2 lines. Leave blank for the proof."
        required: false
        default: ""

permissions:
  contents: write   # only used when publish = branch

jobs:
  generate:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up Node
        uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: Safety guard (max 2 lines unless approved)
        env:
          ONLY: ${{ inputs.only }}
          DRY: ${{ inputs.dry_run }}
          APPROVE: ${{ inputs.approve_more }}
        run: |
          if [ "$DRY" != "false" ]; then echo "Dry run — safe, no spend."; exit 0; fi
          if [ -z "$ONLY" ]; then
            echo "::error::A live run must specify line IDs in 'only'. Refusing to render the whole pack."; exit 1
          fi
          COUNT=$(printf '%s' "$ONLY" | tr ',' '\n' | grep -c .)
          if [ "$COUNT" -gt 2 ] && [ "$APPROVE" != "I APPROVE MORE" ]; then
            echo "::error::$COUNT lines requested but the proof is capped at 2. Set approve_more to 'I APPROVE MORE' to override."; exit 1
          fi
          echo "Approved to generate $COUNT line(s)."

      - name: Generate (or dry run)
        env:
          ELEVENLABS_API_KEY: ${{ secrets.ELEVENLABS_API_KEY }}   # secret, auto-masked, never printed
          VOICE_ID: ${{ inputs.voice_id }}
          ONLY: ${{ inputs.only }}
          LIMIT: ${{ inputs.limit }}
          DRY: ${{ inputs.dry_run }}
          OUT_DIR: ${{ inputs.out_dir }}
        run: |
          set -euo pipefail
          args=( --voice-id "$VOICE_ID" --out "$OUT_DIR" )
          [ -n "$ONLY" ]  && args+=( --only "$ONLY" )
          [ -n "$LIMIT" ] && args+=( --limit "$LIMIT" )
          [ "$DRY" = "false" ] && args+=( --generate )
          echo "node tools/generateVoicePack.mjs ${args[*]}"   # prints args only — never the key
          node tools/generateVoicePack.mjs "${args[@]}"

      - name: List generated files
        if: ${{ inputs.dry_run == 'false' }}
        env:
          OUT_DIR: ${{ inputs.out_dir }}
        run: ls -la "$OUT_DIR" || echo "no output dir (dry run?)"

      - name: Upload MP3s as artifact
        if: ${{ inputs.publish == 'artifact' && inputs.dry_run == 'false' }}
        uses: actions/upload-artifact@v4
        with:
          name: voice-pack-proof
          path: ${{ inputs.out_dir }}/*.mp3
          if-no-files-found: warn

      # Commits the proof files to a SEPARATE branch (never main). For convenience
      # on a tablet, it also flips PREMIUM_VOICE_READY -> true on THIS branch only,
      # so the proof is immediately audible if you preview/serve voice-pack-proof.
      # main is left untouched (premium stays off there until the full pack exists).
      - name: Commit to voice-pack-proof branch
        if: ${{ inputs.publish == 'branch' && inputs.dry_run == 'false' }}
        env:
          OUT_DIR: ${{ inputs.out_dir }}
        run: |
          set -euo pipefail
          git config user.name "voice-pack-bot"
          git config user.email "actions@users.noreply.github.com"
          git checkout -B voice-pack-proof
          if grep -q "const PREMIUM_VOICE_READY = false;" foundations.js 2>/dev/null; then
            sed -i 's/const PREMIUM_VOICE_READY = false;/const PREMIUM_VOICE_READY = true;/' foundations.js
            echo "Enabled premium voice on voice-pack-proof branch."
          else
            echo "::warning::Could not find PREMIUM_VOICE_READY flag in ./foundations.js — set it to true manually on this branch to hear the pack."
          fi
          git add "$OUT_DIR"/*.mp3 tools/voice-pack.lock.json foundations.js
          git commit -m "Voice pack proof (Jack): 2 lines + enable premium on proof branch [skip ci]" || echo "nothing to commit"
          git push -f origin voice-pack-proof
          echo "Pushed to voice-pack-proof. main is untouched."
